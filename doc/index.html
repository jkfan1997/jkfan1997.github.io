# 使用

在main.js中写入以下内容

  import Vue from 'vue'
  import App from './App.vue'
  import VanUI from 'van-ui'
  import "van-ui/dist/van-ui.css"
  Vue.use(VanUI)

  new Vue({
    render: h => h(App),
  }).$mount('#app')
